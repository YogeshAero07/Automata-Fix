# Q.1 Write a program to print as following :
5
54
543
5432
54321

# Ans: 

 Solultion 1:- Using Simple Approach: 

#include <iostream>
using namespace std;

int main() {
	cout<<"5"<<endl;
	cout<<"5"<<"4"<<endl;
	cout<<"5"<<"4"<<"3"<<endl;
	cout<<"5"<<"4"<<"3"<<"2"<<endl;
	cout<<"5"<<"4"<<"3"<<"2"<<"1"<<endl;
	return 0;
}

Solution 2:- 

#include <bits/stdc++.h>
using namespace std;

int main() {

    int num;
    cin>>num;
    
    vector<int> arr;
    
    for(int i=1; i<=num; i++){
        arr.push_back(i);
    }
    
    for(int i=num-1; i>=0; i--){
        for(int j=num-1; j>=i; j--) {
            cout<<arr[j];
        }
        cout<<endl;
    }
     
	return 0;
}
