# Q.1 Write a program to print as following :
5
54
543
5432
54321

# Ans: 

#include <bits/stdc++.h>
using namespace std;

int main() {
    int num;
    cin>>num;
    
    vector<int> arr;
    
    for(int i=1; i<=num; i++){
        arr.push_back(i);
    }
    
    for(int i=num-1; i>=0; i--){
        for(int j=num-1; j>=i; j--) {
            cout<<arr[j];
        }
        cout<<endl;
    }
	return 0;
}

# Q. 4 Write a program to print as following :

1*2*3*4
9*10*11*12
13*14*15*16
5*6*7*8

# Ans: 

#include <bits/stdc++.h>
using namespace std;

int main() {

    for(int i=0; i<13; i=i+4) {
        if(i==4){
            continue;
        } else {
               cout<<i+1<<"*"<<i+2<<"*"<<i+3<<"*"<<i+4;
        cout<<endl;
        }
    }
    cout<<"5"<<"*"<<"6"<<"*"<<"7"<<"*"<<"8";
    
	return 0;
}
